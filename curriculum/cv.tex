  \begin{tabular}{@{}p{\hintscolumnwidth}@{\hspace{\separatorcolumnwidth}}p{\maincolumnwidth}@{}}%

        {\bfseries#3}%

        \ifthenelse{\equal{#6}{}}{}{, #6}%
    }%
  \end{tabular}%
  {\small#7}%
  \par\addvspace{#1}}
